     function ajax_send(frm){

           ​var formData = $("#form").serializeObject();

                 type:'POST',

                 contentType:'application/json',

                 data:JSON.stringify(formData), //{"title":"JSON전송", "contents":"JSON데이타 전송 테스트"} 형태임

                 success: function(data){

                         document.getElementById("rtn").innerHTML = data.trim(); //호출 반환값을 디버깅용 textarea에 출력

                 },error: function(xhr, status, error){

                        document.getElementById("rtn").innerHTML = "[" + xhr.status + "]" + error;

                 } 

          }); 

          return false;

      }
